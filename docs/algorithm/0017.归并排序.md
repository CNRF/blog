## 算法说明

归并排序是一种有效的排序算法，采用分治法（Divide and Conquer）策略。它将数组分成两半，对每一半进行递归排序，然后将两个已排序的部分合并成一个完整的有序数组。

## 递归实现

### 实现思路

1. **分解**：将数组从中间分成两半，分别对左半部分和右半部分进行递归排序。
    - 通过不断分解，最终会将数组分解到每个子数组只包含一个元素。
    - 此时只需要将每个单独的元素视为有序。之后比较两个当个元素大小后进行合并
2. **合并**：将两个已排序的子数组合并成一个有序数组。这个过程需要一个辅助数组来存储合并后的结果，然后再将结果复制回原数组。

```java
package lesson.sort;

import java.util.Arrays;

/**
 * 基本的归并排序
 *
 * @author mark
 *
 */


public class MergeSort {

    /**
     * 递归实现归并排序
     *
     */
    public void recursionMergeSort(int[] arr) {
        if (null == arr || arr.length < 2) {
            return;
        }
        process(arr, 0, arr.length - 1);
    }

    public void process(int[] arr, int left, int right) {
        if (left == right) {
            return;
        }
        int mid = left + ((right - left) >> 1);
        //保证[left,mid]此段数组有序
        process(arr, left, mid);
        process(arr, mid + 1, right);
        merge(arr, left, mid, right);
    }

    //取[left,mid],[mid+1,right]这两个数组进行合并
    public void merge(int[] arr, int left, int mid, int right) {
        int[] help = new int[right - left + 1];
        int i = 0;
        int p1 = left;
        int p2 = mid + 1;
        while (p1 <= mid && p2 <= right) {
            help[i++] = arr[p1] < arr[p2] ? arr[p1++] : arr[p2++];
        }
        //此时要不是p1大于mid或者p2大于right
        //直接复制另外一部分即可
        while (p1 <= mid) {
            help[i++] = arr[p1++];
        }
        while (p2 <= right) {
            help[i++] = arr[p2++];
        }
        for (int h = 0; h < help.length; h++) {
            arr[left + h] = help[h];
        }
    }


    //for test
    public static int[] generateRandomArray(int maxSize, int maxValue) {
        int[] arr = new int[(int) ((maxSize + 1) * Math.random())];
        for (int i = 0; i < arr.length; i++) {
            arr[i] = (int) ((maxValue + 1) * Math.random()) - (int) (maxValue * Math.random());
        }
        return arr;
    }

    // for test
    public static int[] copyArray(int[] arr) {
        if (arr == null) {
            return null;
        }
        int[] res = new int[arr.length];
        for (int i = 0; i < arr.length; i++) {
            res[i] = arr[i];
        }
        return res;
    }

    // for test
    public static boolean isEqual(int[] arr1, int[] arr2) {
        if ((arr1 == null && arr2 != null) || (arr1 != null && arr2 == null)) {
            return false;
        }
        if (arr1 == null && arr2 == null) {
            return true;
        }
        if (arr1.length != arr2.length) {
            return false;
        }
        for (int i = 0; i < arr1.length; i++) {
            if (arr1[i] != arr2[i]) {
                return false;
            }
        }
        return true;
    }

    // for test
    public static void printArray(int[] arr) {
        if (arr == null) {
            return;
        }
        for (int i = 0; i < arr.length; i++) {
            System.out.print(arr[i] + " ");
        }
        System.out.println();
    }

    //for test
    public static void main(String[] args) {
        MergeSort mergeSort = new MergeSort();
        int testTime = 500000;
        int maxSize = 100;
        int maxValue = 100;
        System.out.println("测试开始");
        for (int i = 0; i < testTime; i++) {
            int[] arr1 = generateRandomArray(maxSize, maxValue);
            int[] arr2 = copyArray(arr1);
            mergeSort.recursionMergeSort(arr1);
            //arr2按照大小排序
            Arrays.sort(arr2);
            if (!isEqual(arr1, arr2)) {
                System.out.println("出错了！");
                printArray(arr1);
                printArray(arr2);
                break;
            }
        }
        System.out.println("测试结束");
    }
}

```

## 非递归方式

### 实现思路

非递归实现归并排序的思路是通过迭代的方式来合并数组。
我们从小的子数组开始，逐渐增大子数组的大小，直到整个数组被合并为一个有序数组。

1. **初始化**：设置一个步长（mergeSize），初始值为1
2. **合并过程**：在每次迭代中，将数组分成两个部分，分别为[left, mid]和[mid+1, right]，然后将这两个部分进行合并。
3. **更新步长**：每次迭代后，将步长翻倍
4. **终止条件**：当步长大于数组长度的一半时，停止迭代

### 代码

```java
package lesson.sort;

import java.util.Arrays;

/**
 * 归并排序非递归实现
 *
 * @author mark
 *
 */


public class MergeSortNoProcess {
    public void mergeSort(int[] arr) {
        if (null == arr || arr.length < 2) {
            return;
        }
        int length = arr.length;
        //步长（每次分割原数组长度）
        int mergeSize = 1;
        while (mergeSize < length) {
            int left = 0;
            while (left < length) {
                if (mergeSize >= length - left) {
                    break;
                }
                int mid = left + mergeSize - 1;
                int right = mid + Math.min(mergeSize, length - mid - 1);
                merge(arr, left, mid, right);
                left = right + 1;
            }
            //防止溢出
            if (mergeSize > length / 2) {
                break;
            }
            mergeSize = mergeSize << 1;
        }
    }

    //取[left,mid],[mid+1,right]这两个数组进行合并
    public void merge(int[] arr, int left, int mid, int right) {
        int[] help = new int[right - left + 1];
        int i = 0;
        int p1 = left;
        int p2 = mid + 1;
        while (p1 <= mid && p2 <= right) {
            help[i++] = arr[p1] < arr[p2] ? arr[p1++] : arr[p2++];
        }
        //此时要不是p1大于mid或者p2大于right
        //直接复制另外一部分即可
        while (p1 <= mid) {
            help[i++] = arr[p1++];
        }
        while (p2 <= right) {
            help[i++] = arr[p2++];
        }
        for (int h = 0; h < help.length; h++) {
            arr[left + h] = help[h];
        }
    }

    //for test
    public static int[] generateRandomArray(int maxSize, int maxValue) {
        int[] arr = new int[(int) ((maxSize + 1) * Math.random())];
        for (int i = 0; i < arr.length; i++) {
            arr[i] = (int) ((maxValue + 1) * Math.random()) - (int) (maxValue * Math.random());
        }
        return arr;
    }

    // for test
    public static int[] copyArray(int[] arr) {
        if (arr == null) {
            return null;
        }
        int[] res = new int[arr.length];
        for (int i = 0; i < arr.length; i++) {
            res[i] = arr[i];
        }
        return res;
    }

    // for test
    public static boolean isEqual(int[] arr1, int[] arr2) {
        if ((arr1 == null && arr2 != null) || (arr1 != null && arr2 == null)) {
            return false;
        }
        if (arr1 == null && arr2 == null) {
            return true;
        }
        if (arr1.length != arr2.length) {
            return false;
        }
        for (int i = 0; i < arr1.length; i++) {
            if (arr1[i] != arr2[i]) {
                return false;
            }
        }
        return true;
    }

    // for test
    public static void printArray(int[] arr) {
        if (arr == null) {
            return;
        }
        for (int i = 0; i < arr.length; i++) {
            System.out.print(arr[i] + ",");
        }
        System.out.println();
    }

    //for test
    public static void main(String[] args) {
        MergeSortNoProcess mergeSort = new MergeSortNoProcess();

        int testTime = 500000;
        int maxSize = 100;
        int maxValue = 100;
        System.out.println("测试开始");
        for (int i = 0; i < testTime; i++) {
            int[] arr1 = generateRandomArray(maxSize, maxValue);
            int[] arr2 = copyArray(arr1);
            try {
                mergeSort.mergeSort(arr1);
            } catch (Exception e) {
                System.out.println(e.toString());
                printArray(arr1);
            }
            //arr2按照大小排序
            Arrays.sort(arr2);
            if (!isEqual(arr1, arr2)) {
                System.out.println("出错了！");
                printArray(arr1);
                printArray(arr2);
                break;
            }
        }
        System.out.println("测试结束");
    }
}
```
